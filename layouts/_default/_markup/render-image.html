{{- /* Bundle-local jpg/png get resized to 800px max as WebP with 2x srcset; absolute/external URLs and other formats pass through. Markdown title slot doubles as a class hint when prefixed with `.` — e.g. ![alt](src ".right wide") → class="right wide". */ -}} {{- $src := .Destination -}} {{- $alt := .PlainText -}} {{- $classes := "" -}} {{- $titleAttr := "" -}} {{- with .Title -}} {{- if hasPrefix . "." -}} {{- $classes = strings.TrimPrefix "." . -}} {{- else -}} {{- $titleAttr = . -}} {{- end -}} {{- end -}} {{- $extra := "" -}} {{- with $classes }}{{ $extra = printf " class=%q" . }}{{ end -}} {{- with $titleAttr }}{{ $extra = printf "%s title=%q" $extra . }}{{ end -}} {{- if or (hasPrefix $src "http") (hasPrefix $src "/") -}} {{- else -}} {{- $img := .Page.Resources.GetMatch $src -}} {{- if $img -}} {{- $ext := lower (strings.TrimPrefix "." (path.Ext $src)) -}} {{- if in (slice "jpg" "jpeg" "png") $ext -}} {{- $maxW := 800 -}} {{- $w1x := cond (gt $img.Width $maxW) $maxW $img.Width -}} {{- $w2x := cond (gt $img.Width (mul $maxW 2)) (mul $maxW 2) $img.Width -}} {{- $1x := $img.Resize (printf "%dx webp q85" $w1x) -}} {{- $2x := $img.Resize (printf "%dx webp q85" $w2x) -}} {{- else -}} {{- end -}} {{- else -}} {{- end -}} {{- end -}}
